function findPos(obj) {
    // this sometimes causes an error with nulls (see: http://37signals.com/svn/)
    // i think it's just when no #comments element is found.
	var curleft = curtop = 0;
    if (obj.offsetParent) {
        do {
    	    curleft += obj.offsetLeft;
    		curtop += obj.offsetTop;
    	} while (obj = obj.offsetParent);
    	return [curleft,curtop];
	}
};

var comElement = document.getElementById('comments') || document.getElementById('dsq-comments');
var comHeight = findPos(comElement)[1];
var winHeight = window.innerHeight;
var docHeight = document.body.clientHeight;

var regionsDiv = document.createElement('div');
regionsDiv.id = 'readingRegions';
regionsDiv.style.cssText = "height: "+winHeight+"px; width: 10px; position: fixed; top: 0px; right: 0px;";

var articleDiv = document.createElement('div');
articleDiv.id = 'readingRegionArticle';
articleDiv.style.cssText = "background-color: green; height: "+(winHeight*comHeight/docHeight)+"px; width: 10px;";

var commentsDiv = document.createElement('div');
commentsDiv.id = 'readingRegionComments';
commentsDiv.style.cssText = "background-color: red; height: "+(winHeight*(docHeight-comHeight)/docHeight)+"px; width: 10px;";

regionsDiv.appendChild(articleDiv);
regionsDiv.appendChild(commentsDiv);

document.body.appendChild(regionsDiv);
